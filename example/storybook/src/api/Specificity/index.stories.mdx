## API

```jsx
import React from 'react';
import { View } from 'react-native';
import { styled } from '@dank-style/react';

const StyledButton = styled(
  View,
  {
    baseStyle: {
      style: {
        borderRadius: 4,
        flexDirection: 'row',
        justifyContent: 'center',
        alignItems: 'center',
        outlineWidth: 0,
        m: 12,
      },
    },
    variants: {
      bluebox: {
        style: {
          bg: '$blue400',
        },
      },
    },
    sizes: {
      sm: {
        style: {
          px: '$3',
          py: '$2',
        },
      },
      md: {
        style: {
          px: '$4',
          py: '$3',
        },
      },
    },
    defaultProps: {
      size: 'md',
      variant: 'bluebox',
    },
  },
  {}
);
```
